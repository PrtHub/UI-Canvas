```ts
import { CanvasColorPicker } from "@/components/ui/canvas-color-picker";
```

## Usage

```tsx
import { CanvasColorPicker } from "@/components/ui/canvas-color-picker";

export default function ColorPickerExample() {
  return (
    <CanvasColorPicker
      defaultValue="#4f46e5"
      defaultConsistency="medium"
      previewSize="md"
      onChange={(color) => console.log("Selected color:", color)}
    />
  );
}
```

## Props

### Canvas Color Picker

| Prop name          | Type                          | Default   | Description                                   |
| ------------------ | ----------------------------- | --------- | --------------------------------------------- |
| value              | string                        | -         | Controlled value for the color picker         |
| defaultValue       | string                        | "#000000" | Initial color value                           |
| defaultConsistency | "thin" \| "medium" \| "thick" | "medium"  | Initial paint consistency                     |
| previewSize        | "sm" \| "md" \| "lg"          | "md"      | Size of the color preview area                |
| showBrushPreview   | boolean                       | true      | Show brush stroke preview with selected color |
| onChange           | (color: string) => void       | -         | Callback when color changes                   |
| className          | string                        | -         | Additional CSS classes                        |

### Paint Consistency Levels

| Level  | Line Width | Opacity | Description                               |
| ------ | ---------- | ------- | ----------------------------------------- |
| Light  | 2px        | 30%     | Thin lines with subtle color application  |
| Medium | 4px        | 60%     | Balanced lines with moderate opacity      |
| Heavy  | 8px        | 90%     | Thick lines with strong color application |
