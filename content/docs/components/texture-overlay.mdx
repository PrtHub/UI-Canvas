## Usage

```tsx
import { TextureOverlay } from "@/components/ui/texture-overlay";

export default function TextureOverlayExample() {
  return (
    <TextureOverlay
      texture="glass"
      intensity="light"
      blendMode="screen"
      className="rounded-xl bg-gradient-to-br from-white/40 to-white/10 p-6"
    >
      <h2 className="text-2xl font-bold">Glass Effect</h2>
      <p className="mt-2">Content with modern glass texture</p>
    </TextureOverlay>
  );
}
```

## Props

### Texture Overlay

| Prop name | Type                                                                                                                    | Default      | Description                                                |
| --------- | ----------------------------------------------------------------------------------------------------------------------- | ------------ | ---------------------------------------------------------- |
| texture   | "watercolor" \| "oil" \| "charcoal" \| "canvas" \| "noise" \| "grain" \| "paper" \| "glass" \| "frost" \| "holographic" | "watercolor" | The texture style to apply                                 |
| intensity | "light" \| "medium" \| "heavy"                                                                                          | "medium"     | The intensity of the texture effect                        |
| blendMode | "multiply" \| "screen" \| "overlay" \| "soft-light" \| "hard-light" \| "color-dodge" \| "color-burn"                    | "soft-light" | The blend mode for the texture                             |
| animate   | boolean                                                                                                                 | false        | Enable hover animation that reveals the content underneath |
| className | string                                                                                                                  | -            | Additional CSS classes                                     |
