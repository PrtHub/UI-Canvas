## Usage

```tsx
import { PatternLibrary } from "@ui-canvas/components";

export default function PatternDemo() {
  const [currentPattern, setCurrentPattern] = useState({
    id: "1",
    name: "Dots Grid",
    type: "dots",
    size: 4,
    spacing: 20,
    rotation: 0,
    opacity: 0.5,
    color: "#6366f1",
  });

  return (
    <PatternLibrary
      value={currentPattern}
      onValueChange={setCurrentPattern}
      patterns={patterns}
      onPatternsChange={setPatterns}
    />
  );
}
```

## Props

| Prop               | Type                            | Description                                |
| ------------------ | ------------------------------- | ------------------------------------------ |
| `value`            | `Pattern`                       | The currently selected pattern             |
| `onValueChange`    | `(pattern: Pattern) => void`    | Callback when the selected pattern changes |
| `patterns`         | `Pattern[]`                     | Array of available patterns                |
| `onPatternsChange` | `(patterns: Pattern[]) => void` | Callback when patterns array is modified   |

## Types

```tsx
interface Pattern {
  id: string;
  name: string;
  type:
    | "dots"
    | "grid"
    | "lines"
    | "circles"
    | "squares"
    | "triangles"
    | "diamonds";
  size: number;
  spacing: number;
  rotation: number;
  opacity: number;
  color: string;
}
```

## Features

The Pattern Library component provides a comprehensive set of features for managing and applying patterns:

- **Pattern Types**: Choose from 6 different pattern types (dots, grid, lines, circles, squares, triangles, diamonds)
- **Pattern Customization**: Adjust size, spacing, rotation, and opacity
- **Color Control**: Choose from predefined colors or use custom colors
- **Pattern Management**: Add, duplicate, and delete patterns
- **Visual Preview**: See pattern changes in real-time
- **Dark Mode Support**: Seamless integration with light and dark themes

## Examples

### Basic Pattern Library

```tsx
import { PatternLibrary } from "@ui-canvas/components";

export function BasicPatternLibrary() {
  const [pattern, setPattern] = useState({
    id: "1",
    name: "Basic Grid",
    type: "grid",
    size: 1,
    spacing: 20,
    rotation: 0,
    opacity: 0.3,
    color: "#64748b",
  });

  return <PatternLibrary value={pattern} onValueChange={setPattern} />;
}
```

### Custom Pattern Types

```tsx
import { PatternLibrary } from "@ui-canvas/components";

const customPatterns = [
  {
    id: "custom-1",
    name: "Diamond Grid",
    type: "diamonds",
    size: 8,
    spacing: 24,
    rotation: 0,
    opacity: 0.4,
    color: "#ec4899",
  },
  // Add more custom patterns...
];

export function CustomPatternLibrary() {
  const [patterns, setPatterns] = useState(customPatterns);
  const [current, setCurrent] = useState(patterns[0]);

  return (
    <PatternLibrary
      value={current}
      onValueChange={setCurrent}
      patterns={patterns}
      onPatternsChange={setPatterns}
    />
  );
}
```
