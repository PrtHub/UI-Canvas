## Usage

### Basic Example

```tsx
import { useState } from "react";
import BrushStyleSelector, {
  BrushPreset,
} from "@/components/ui-canvas/brush-style-selector";

export function SimpleBrushSelector() {
  const [brush, setBrush] = useState<BrushPreset>({
    id: "1",
    name: "Default Brush",
    size: 10,
    opacity: 1,
    style: "round",
  });

  return <BrushStyleSelector value={brush} onValueChange={setBrush} />;
}
```

### Advanced Example

```tsx
import { useState } from "react";
import BrushStyleSelector, {
  BrushPreset,
} from "@/components/ui-canvas/brush-style-selector";

export function AdvancedBrushSelector() {
  const presets: BrushPreset[] = [
    {
      id: "1",
      name: "Pencil",
      size: 2,
      opacity: 1,
      style: "round",
      hardness: 100,
      spacing: 1,
    },
    {
      id: "2",
      name: "Soft Brush",
      size: 20,
      opacity: 0.8,
      style: "round",
      color: "#3b82f6",
      hardness: 50,
      spacing: 1.2,
    },
    {
      id: "3",
      name: "Calligraphy",
      size: 12,
      opacity: 1,
      style: "round",
      color: "#6366f1",
      hardness: 90,
      angle: 45,
      spacing: 1,
    },
  ];

  const [currentBrush, setCurrentBrush] = useState<BrushPreset>(presets[0]);

  return (
    <BrushStyleSelector
      value={currentBrush}
      presets={presets}
      onValueChange={setCurrentBrush}
      showAdvancedControls={true}
    />
  );
}
```

## Props

| Name                   | Type                           | Default     | Description                       |
| ---------------------- | ------------------------------ | ----------- | --------------------------------- |
| `value`                | `BrushPreset`                  | Required    | The current brush settings        |
| `presets`              | `BrushPreset[]`                | `[]`        | Array of predefined brush presets |
| `onValueChange`        | `(value: BrushPreset) => void` | `undefined` | Called when brush settings change |
| `minSize`              | `number`                       | `1`         | Minimum brush size                |
| `maxSize`              | `number`                       | `100`       | Maximum brush size                |
| `showAdvancedControls` | `boolean`                      | `false`     | Show advanced brush controls      |
| `className`            | `string`                       | `undefined` | Additional CSS classes to apply   |

## Types

### BrushPreset

```typescript
type BrushStyle = "round" | "square" | "star" | "custom";

interface BrushPreset {
  id: string; // Unique identifier for the preset
  name: string; // Display name of the preset
  size: number; // Brush size in pixels
  opacity: number; // Brush opacity (0-1)
  style: BrushStyle; // Brush shape style
  color?: string; // Optional brush color
  hardness?: number; // Brush edge softness (0-100)
  angle?: number; // Brush angle in degrees (0-360)
  spacing?: number; // Brush spacing multiplier (0.1-2)
}
```
